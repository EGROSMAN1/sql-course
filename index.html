SELECT c.company,
    COUNT(o.id) AS orders,
    SUM(od.quantity * od.unit_price)
        AS total_revenue,
    RANK() OVER (
      ORDER BY SUM(...) DESC
    ) AS rank_pos
FROM customers c
JOIN orders o ON c.id = o.customer_id
JOIN order_details od ON ...
GROUP BY c.company
LIMIT 10;
